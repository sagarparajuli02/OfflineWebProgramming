      <script>
         $(function() {

            var $doc = $(document),
            $win = $(window);

            var $intro = $('.intro'),
            $items = $intro.find('.item'),
            itemsLen = $items.length,
				
            svgs = $intro.find('svg').drawsvg({
               callback: animateIntro,
               easing: 'easeOutQuart'
            }),
				
            currItem = 0;

            function animateIntro() {
               $items.removeClass('active').eq( currItem++ % itemsLen 
                  ).addClass('active').find('svg').drawsvg('animate');
            }

            animateIntro();

            var $header = $('header'),
            headerOffTop = $header.offset().top,
            isFixed = false;

            function menu() {
               if ( $win.scrollTop() >= headerOffTop ) {
                  if ( !isFixed ) {
                     isFixed = true;
                     $header.addClass('affix');
                  }
               } else if ( isFixed ) {
                  isFixed = false;
                  $header.removeClass('affix');
               }
            }

            $win.on('scroll', menu);
            menu();

            $header.on('click', 'a[href^="#"]', function(e) {
               e.preventDefault();

               var hash = this.hash,
               offset = $(hash).offset().top;

               $('body, html').animate({
                  scrollTop: offset
               }, 600, 'easeInOutQuart', function() {
                  document.location.hash = hash;
               });
            });

         });
      </script>
